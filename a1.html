code snippet

// set integer named time equal to 1000 (1 second using delay function)
int time = 1000;

// set integer named blue equal to 4 (blue led is connected to pin 4)
int blue = 4;
// set integer named white equal to 8 (white led is connected to pin 8)
int white = 8;
// set integer named green equal to 12 (green led is connected to pin 12)
int green = 12;

// run once at the start
void setup() {
  // set the pin connected to the blue led (pin 4) to an output
  pinMode(blue, OUTPUT);
  // set the pin connected to the white led (pin 8) to an output
  pinMode(white, OUTPUT);
  // set the pin connected to the green led (pin 12) to an output
  pinMode(green, OUTPUT);
}

// repeat infinitely
void loop() {
  // set white led to on (5 V)
  digitalWrite(white, HIGH);
  // set green led to off (0 V)
  digitalWrite(green, LOW);

  // wait 1 second (time = 1000)
  delay(time);

  // set green led to on (5 V)
  digitalWrite(green, HIGH);
  // set blue led to off (0 V)
  digitalWrite(blue, LOW);

  // wait 1 second (time = 1000)
  delay(time);

  // set blue led to on (5 V)
  digitalWrite(blue, HIGH);
  // set white led to off (0 V)
  digitalWrite(white, LOW);

  // wait 1 second (time = 1000)
  delay(time);
}
